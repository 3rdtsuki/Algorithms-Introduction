### 快速傅里叶变换

https://www.luogu.com.cn/problem/P3803

```cpp
#include <iostream>
#include <cmath>
#include <complex>
using namespace std;
const int maxn = 2005;
const double pi=acos(-1);
complex<double>a[maxn],b[maxn];

void FFT(complex<double>*a,int n,int op){
	if(n==0)return;
	complex<double>a1[n],a2[n];
	for(int i=0;i<n;++i){
		a1[i]=a[2*i];
		a2[i]=a[2*i+1];
	}
	FFT(a1,n/2,op);
	FFT(a2,n/2,op);
	complex<double>wn(cos(pi/n),sin(pi/n)*op);
	complex<double>w(1,0);
	for(int i=0;i<n;++i,w*=wn){
		a[i]=a1[i]+w*a2[i];
		a[i+n]=a1[i]-w*a2[i];
	}
	
}
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<=n;++i){
		cin>>a[i];
	}
	for(int i=0;i<=m;++i){
		cin>>b[i];
	}
	m=m+n;
	for(n=1;n<=m;n<<=1);//找到比最小的比m+n大的2的幂
	FFT(a,n/2,1);
	FFT(b,n/2,1);
	for(int i=0;i<n;++i){
		a[i]*=b[i];
	}
	FFT(a,n/2,-1);
	for(int i=0;i<=m;++i){
		cout<<fabs(a[i].real()/n)<<' ';
	}
}
```
