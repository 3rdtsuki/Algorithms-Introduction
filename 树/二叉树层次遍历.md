### bfs，关键是last和newlast记录每层最右节点换行
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
#include <stack>
#include <queue>
using namespace std;
const int maxn = 2005;
struct binary_node{
    int ls;
    int rs;
}node[maxn];
int bfs(int root){
    queue<int>q;
    q.push(root);
    int last=root; //last记录每层最右边的节点
    int newlast=root; //newlast记录新加入的节点
    while(!q.empty()){
        int father=q.front();
        q.pop();
        cout<<father<<' ';
        if(node[father].ls){
            q.push(node[father].ls);
            newlast=node[father].ls;
        }
        if(node[father].rs){
            q.push(node[father].rs);
            newlast=node[father].rs;
        }
        if(last==father){ //最右节点作为根输出时，要换行了
            cout<<endl;
            last=newlast; //正好将下一层最右节点更新为该层最右节点的儿子
        }
    }
}
signed main(){
    int n;
    cin>>n;
    for(int i=1;i<=n;++i){
        cin>>node[i].ls>>node[i].rs;
    }
    bfs(1);
}
```
